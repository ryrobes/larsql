# A Cascade with Candidates + Wards
cascade_id: fraud_assessment
inputs_schema:
  transaction_id: "Transaction ID"
  amount: "Transaction amount"

cells:
  - name: assess_risk
    instructions: |
      Analyze transaction for fraud risk.
      Return JSON: {risk_score, risk_level, reasoning}

    # Run 3 parallel candidates, pick best
    candidates:
      factor: 3
      mode: evaluate
      evaluator_instructions: Pick the most conservative assessment

    # Validate output contract
    wards:
      post:
        - validator:
            json_schema: risk_assessment_schema.json
          mode: retry
          max_retries: 2

    handoffs: [make_decision]
-- Call Cascades from SQL as UDFs
SELECT
  transaction_id,
  amount,
  rvbbit_cascade_udf(
    'traits/fraud_assessment.yaml',
    json_object('transaction_id', transaction_id, 'amount', amount)
  ) AS fraud_check
FROM pending_transactions
WHERE amount > 1000;

-- Query Receipts for cost analysis
SELECT
  cell_name,
  AVG(cost_usd) AS avg_cost,
  AVG(duration_ms) AS avg_latency,
  cache_hit_rate
FROM rvbbit_receipts
WHERE cascade_id = 'fraud_assessment'
GROUP BY cell_name;
# Map Cascade over array inputs
import rvbbit

transactions = [
    {"transaction_id": 1, "amount": 150},
    {"transaction_id": 2, "amount": 99999},
]

results = rvbbit.map_cascade(
    cascade="traits/fraud_assessment.yaml",
    inputs=transactions,
    concurrency=8,
    cache="aggressive",
    receipts=True,
)

# Each result includes outputs + receipts
for r in results:
    print(r.output, r.receipt.cost_usd, r.receipt.duration_ms)
